Diagramas de secuencia
1)préstamo de juegos
@startuml
actor Cliente
participant Cafe
participant InventarioJuegos
participant Mesa
participant Juego
participant Mesero
participant Prestamo

Cliente -> Cafe : solicitarPrestamo(juego, mesa)

Cafe -> InventarioJuegos : verificarDisponibilidad(juego)
Cafe -> Mesa : validarCapacidad()
Cafe -> Juego : validarRestricciones()

alt Juego dificil
    Cafe -> Mesero : verificarDisponibilidad()
end

Cafe -> Prestamo : crearPrestamo()
Prestamo -> InventarioJuegos : reducirStock()

Cafe -> Cliente : confirmarPrestamo()

@enduml

2)Venta de producto
@startuml
actor Cliente
participant Cafe
participant InventarioCafeteria
participant Venta
participant DetalleVenta

Cliente -> Cafe : realizarVenta(productos)

Cafe -> InventarioCafeteria : verificarStock()

loop por cada producto
    Cafe -> DetalleVenta : crearDetalle()
end

Cafe -> Venta : crearVenta()
Venta -> Venta : calcularImpuestos()
Venta -> Venta : calcularPropina()

Cafe -> InventarioCafeteria : actualizarStock()

Cafe -> Cliente : confirmarVenta()

@enduml

3) cambio de turno
@startuml
actor Empleado
participant Cafe
participant SolicitudCambioTurno
participant Administrador

Empleado -> Cafe : solicitarCambioTurno()

Cafe -> SolicitudCambioTurno : crearSolicitud()

Cafe -> Administrador : notificarSolicitud()

Administrador -> Cafe : aprobar/rechazar()

Cafe -> Empleado : notificarResultado()

@enduml
